        ComplexNumber first = new ComplexNumber(0, 0);
        ComplexNumber second = new ComplexNumber(0, 0);

        while (true) {
            System.out.println("Hello Men!");

            first = parseComplexNumber();
            String op = parseOperator();
            second = parseComplexNumber();

            switch (op) {
                case "+":
                    first.add(second);
                    break;
                    case "-":
                        first.subtract(second);
                        break;
                        case "*":
                            first.multiply(second);
                            break;
                            case "/":
                                first.divide(second);
                                break;
                                default:
                                    continue;
            }

            System.out.println("Result: " + first);
        }

    private static ComplexNumber parseComplexNumber() {
        Scanner in = new Scanner(System.in);
        System.out.print("Enter the real and imaginary parts of a complex number: ");
        float real = in.nextFloat();
        float imaginary = in.nextFloat();
        return new ComplexNumber(real, imaginary);
    }

    private static String parseOperator() {
        Scanner in = new Scanner(System.in);
        System.out.print("Enter the operator: ");
        String operator = in.nextLine();
        return operator;
    }
